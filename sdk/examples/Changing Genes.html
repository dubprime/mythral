Changing Genes

Import AgentClient from module ia.gaius.agent_client

[1]:

from ia.gaius.agent_client import AgentClient

[2]:

agent_info = {'name': '',
              'domain': 'gaius-api',
              'secure': False,
              'api_key': 'ABCD-1234'}

[3]:

agent = AgentClient(agent_info)

[4]:

agent.connect()

[4]:

{'connection': 'okay', 'agent': 'simple'}

[5]:

agent.show_status()

[5]:

 'PREDICT': True,

agent.get_all_genes(nodes=['P1'])
